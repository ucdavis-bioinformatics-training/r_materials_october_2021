if (!("BiocManager" %in% rownames(installed.packages()))){
  install.packages("BiocManager")
}

  install.packages("ggplot2")
}
if (!("tidyr" %in% rownames(installed.packages()))){
  install.packages("tidyr")
}
if (!("dplyr" %in% rownames(installed.packages()))){

}
